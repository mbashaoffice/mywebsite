    public class FileNameExtractor {
    public static void main(String[] args) {
        String input = "test1_CAS00000281_2025-02-09132300027.pdf"; // Example input
        System.out.println("Extracted File Name: " + extractFileName(input));
    }

    public static String extractFileName(String input) {
        // Check if input is null or empty
        if (input == null || input.isEmpty()) {
            return "Invalid filename";
        }

        // Find the last dot (.) to get the extension dynamically
        int lastDotIndex = input.lastIndexOf(".");
        String extension = (lastDotIndex != -1 && lastDotIndex > 0) ? input.substring(lastDotIndex) : "";

        // Find the first underscore (_)
        int firstUnderscoreIndex = input.indexOf("_");
        
        // Extract everything before the first "_" and add back the extension
        if (firstUnderscoreIndex != -1) {
            return input.substring(0, firstUnderscoreIndex) + extension;
        }

        // If no underscore is found, return the original filename
        return input;
    }
}
